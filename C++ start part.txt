#include <iostream>
using namespace std;
int main()
{
    int n;
    int d,dem,tong;
    dem = 0;
    tong = 0;
    cin >> n;
    int a[1000];
    for (int i = 0; i < n; i++)
    {
        cin >> a[i];
    }
    for (int i = 0; i < n; i++)
    {
        d = 0;
        if (a[i] != 1)
        {
            for (int j = 2; j < a[i]; j++)
            {
                if (a[i] % j == 0)
                    d = d + 1;
            }
            if (d == 0)
            {
                dem = dem + 1;
                tong = tong + a[i];
            }
        }
    }
    cout << "so luong so nguyen to" << " " << dem << endl;
    cout << "tong cac so nguyen to" << " " << tong << endl;
    return 0;
}
